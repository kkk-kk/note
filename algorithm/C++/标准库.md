## [sort](https://www.cnblogs.com/AlvinZH/p/6784862.html?utm_source=itdadao&utm_medium=referral)

+ `#include <algorithm>`

+ `sort(first_pointer,first_pointer+n,cmp) ` 默认升序
+ `stable_sort()`是稳定的

自己编写排序函数

```c++
bool compare(int a,int b){
	return a>b;  // 降序
}
int main(){
	int a[10] = {1,3,...};
	sort(a,a+10,compare); // 不需要传参
}
```

自定义结构体

```c++
typedef struct Node {
	int a, b;
}Node;
Node arr[100];
bool compare(Node x, Node y) {  //先按a值升序排列，如果a值相同，再按b值降序排列
	if (x.a != y.a) return x.a < y.a;
	return x.b < y.b;
}
```

例题

```c++
#include <iostream>
#include <algorithm>
#include <string>
#include <vector>
using namespace std; 

bool cmp(char a, char b) {
	int x = a, y = b;
	if (a >= 'A' && a <= 'Z')
		x += 32;
	if (b >= 'A' && b <= 'Z')
		y += 32;
	return x <= y;
}

int main() {
	string str,str2;
	int i, c = 0;
	while (getline(cin, str)) {
		for (i = 0;i < str.size();i++) {
			if ((str[i] >= 'A' && str[i] <= 'Z') || (str[i] >= 'a' && str[i] <= 'z'))
				str2+=str[i];
		}
		sort(str2.begin(), str2.end(), cmp);
		c = 0;
		for (i = 0;i < str.size();i++) {
			if ((str[i] >= 'A' && str[i] <= 'Z') || (str[i] >= 'a' && str[i] <= 'z'))
				cout << str2[c++];
			else cout << str[i];
		}
		str = "";
		str2 = "";
		cout << endl;
	}

}
```
