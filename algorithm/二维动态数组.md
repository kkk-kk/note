### 二维动态数组

#### 指针数组

```c++
#include <iostream>
#include <stdlib.h>
using namespace std;
int main()
{
    int row,column;
    cin >> row >> column;
    //申请空间
    int **a =  new int* [row];
    for(int i = 0; i < row; i++)
        a[i] = new int[column];

    //释放空间
    for(int i = 0; i < row; i++)
        delete []a[i];
    a = NULL;
    return 0;
}
```
