```c++
bool Del_s_t(SqList &L,ElemType s,ElemType t){
	if(L.length == 0) {        // 线性表为空
        cout << "线性表为空" << endl;
        return false;
	}
    if(s >= t){
        cout << "s t 不合理" << endl;
        return false
    }
    int i,j;
    for(i = 0;i<L.length && L.data[i]<s;i++);   // 寻找值大于等于s的第一个元素
    if(i >= L.length)
        return false;
    for(j=1;j<L.length && L.data[j]<=t;j++);    // 寻找值大于t的第一个元素
    for(;j<L.length;i++,j++)
        L.data[i] = L.data[j];
    L.length = i;
    return true;
}
```
